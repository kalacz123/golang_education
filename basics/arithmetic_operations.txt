package main

import "fmt"

func main() {
	var a, b int = 10, 3
	var result int

	result = a + b
	fmt.Println("Addition:", result)

	result = a - b
	fmt.Println("Subtraction:", result)

	result = a * b
	fmt.Println("Multiplication:", result)

	result = a / b
	fmt.Println("Division:", result)

	const p float64 = 22 / 7
	fmt.Println("Value of p:", p)

	// Overflow with signed integers
	var maxInt int64 = 9223372036854775807
	fmt.Println("Max Int:", maxInt)

	maxInt = maxInt + 1 // Will be negative cause of int64 overflow
	fmt.Println("Max Int:", maxInt)

	// Overflow with unsigned integers
	var uMaxInt uint64 = 18446744073709551615
	fmt.Println("Max Uint:", uMaxInt)

	uMaxInt = uMaxInt + 1 // Will be 0 cause of uint64 overflow
	fmt.Println("Max Uint:", uMaxInt)
}
